project structure  
    .  
    ├── config                  # Configuration file for "config-lite"  
    ├── lib                     # Database design

    ├── models                  # The files that hold the operating database  
    ├── public                  # Static files  
    ├── routes                  # Route files  
    ├── views                   # Template file  
    ├── app.js                  # Express configuration   
    └── www                     # The entrance
